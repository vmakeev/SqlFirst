/// <summary>
/// Добавляет параметр нестандартного типа к команде
/// </summary>
/// <param name="command">Команда SQL</param>
/// <param name="typeName">Тип параметра</param>
/// <param name="parameterName">Имя параметра</param>
/// <param name="value">Значение параметра</param>
/// <param name="length">Длина параметра</param>
protected virtual void AddCustomParameter(IDbCommand command, string typeName, string parameterName, object value, int? length = null)
{
	var parameter = new TestCommandParameterType
	{
		ParameterName = parameterName,
		TestCommandParameterSpecificCustomDbTypePropertyName = typeName,
		Value = value ?? DBNull.Value
	};

	if (length.HasValue && length.Value > 0)
	{
		parameter.Size = length.Value;
	}
	
	command.Parameters.Add(parameter);
}
