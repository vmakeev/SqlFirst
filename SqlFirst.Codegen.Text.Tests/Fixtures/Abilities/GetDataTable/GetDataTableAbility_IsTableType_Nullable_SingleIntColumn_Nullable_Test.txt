/// <summary>
/// Создает DataTable для типа SomeCustomTableType
/// </summary>
/// <param name="values">Набор данных, который будет помещен в DataTable</param>
protected virtual DataTable GetSomeCustomTableTypeDataTable(IEnumerable<int?> values)
{
	var dataTable = new DataTable();
	dataTable.Columns.Add("some_int_column", typeof(int?));

	if (values == null)
	{
		return null;
	}

	foreach (int? value in values)
	{
		DataRow dataRow = dataTable.NewRow();
		dataRow["some_int_column"] = value ?? DBNull.Value;
		dataTable.Rows.Add(dataRow);
	}

	return dataTable;
}
