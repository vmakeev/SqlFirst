/// <summary>
/// Создает DataTable для типа SomeCustomTableType
/// </summary>
/// <param name="values">Набор данных, который будет помещен в DataTable</param>
protected virtual DataTable GetSomeCustomTableTypeDataTable(IEnumerable<CustomItem> values)
{
	var dataTable = new DataTable();
	dataTable.Columns.Add("some_nvarchar_column", typeof(string));
	dataTable.Columns.Add("some_uuid", typeof(Guid));

	if (values == null)
	{
		return null;
	}

	foreach (CustomItem value in values)
	{
		DataRow dataRow = dataTable.NewRow();
		dataRow["some_nvarchar_column"] = value.SomeNvarcharColumn ?? throw new ArgumentException("Type [SomeCustomTableType] does not allow null values at column [some_nvarchar_column]");
		dataRow["some_uuid"] = value.SomeUuid;
		dataTable.Rows.Add(dataRow);
	}

	return dataTable;
}
